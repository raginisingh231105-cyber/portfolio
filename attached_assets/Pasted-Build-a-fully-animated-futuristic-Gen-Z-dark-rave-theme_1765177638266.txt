7. BARBA.JS PAGE TRANSITIONS

Use:

data-barba="wrapper"
data-barba="container"


In /lib/barba.ts:

import barba from '@barba/core'
import gsap from 'gsap'

export default function initBarba() {
  barba.init({
    transitions: [
      {
        name: 'rave',
        leave({ current }) {
          return gsap.to(current.container, {
            opacity: 0,
            y: -40,
            duration: 0.5
          })
        },

        enter({ next }) {
          return gsap.from(next.container, {
            opacity: 0,
            y: 40,
            duration: 0.5
          })
        }
      }
    ]
  })
}


Initialize inside _app.tsx.
